CREATE DATABASE IF NOT EXISTS dbsapa;
USE dbsapa;

SET GLOBAL max_allowed_packet = 64*1024*1024;

CREATE TABLE AdminSapa(
    admin_id INT AUTO_INCREMENT PRIMARY KEY,
    fullname VARCHAR(50) NOT NULL,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(50) NOT NULL
);

CREATE TABLE SchoolCoordinator(
    school_coordinator_id INT AUTO_INCREMENT PRIMARY KEY,
    school_coordinator_profile LONGBLOB,
    email VARCHAR(50) NOT NULL UNIQUE,
    firstname VARCHAR(50) NOT NULL,
    middlename VARCHAR(50) NOT NULL,
    lastname VARCHAR(50) NOT NULL,
    phone_number VARCHAR(50) UNIQUE,
    birthdate DATE,
    address TEXT,
    gender ENUM('Male', 'Female', 'Other') DEFAULT 'Other',
    bio TEXT,
    password VARCHAR(50) NOT NULL,
    is_approved BOOLEAN DEFAULT FALSE,
    admin_id INT,
    FOREIGN KEY (admin_id) REFERENCES AdminSapa(admin_id) ON DELETE CASCADE
);

CREATE TABLE HospitalCoordinator(
    hospital_coordinator_id INT AUTO_INCREMENT PRIMARY KEY,
    hospital_coordinator_profile LONGBLOB,
    email VARCHAR(50) NOT NULL UNIQUE,
    firstname VARCHAR(50) NOT NULL,
    middlename VARCHAR(50) NOT NULL,
    lastname VARCHAR(50) NOT NULL,    
    phone_number VARCHAR(50) UNIQUE,
    birthdate DATE,
    address TEXT,
    gender ENUM('Male', 'Female', 'Other') DEFAULT 'Other',
    bio TEXT,
    password VARCHAR(50) NOT NULL,
    is_approved BOOLEAN DEFAULT FALSE,
    admin_id INT,
    FOREIGN KEY (admin_id) REFERENCES AdminSapa(admin_id) ON DELETE CASCADE
);

CREATE TABLE HospitalPost (
    hospital_post_id INT AUTO_INCREMENT PRIMARY KEY,
    hospital_coordinator_id INT NOT NULL,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    imageFile LONGBLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (hospital_coordinator_id) REFERENCES HospitalCoordinator(hospital_coordinator_id) ON DELETE CASCADE
);


CREATE TABLE School(
    school_id INT AUTO_INCREMENT PRIMARY KEY,
    school_profile LONGBLOB NOT NULL,
    school_name VARCHAR(255) UNIQUE NOT NULL,
    school_number VARCHAR(255) UNIQUE NOT NULL,
    school_code VARCHAR(255),
    school_email VARCHAR(50) UNIQUE NOT NULL,
    school_address TEXT,
    school_bio TEXT,
    school_coordinator_id INT,
    is_approved BOOLEAN DEFAULT FALSE,
    admin_id INT,
    approval_date TIMESTAMP NULL DEFAULT NULL,
    FOREIGN KEY (school_coordinator_id) REFERENCES SchoolCoordinator(school_coordinator_id) ON DELETE CASCADE,
    FOREIGN KEY (admin_id) REFERENCES AdminSapa(admin_id) ON DELETE SET NULL
);


CREATE TABLE Student (
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    student_profile LONGBLOB,
    student_firstname VARCHAR(255),
    student_middlename VARCHAR(255),
    student_lastname VARCHAR(255),
    student_address TEXT,
    phone_number VARCHAR(20) UNIQUE,
    student_email VARCHAR(255) UNIQUE,
    student_birthdate DATE,
    student_gender ENUM('Male', 'Female', 'Other') DEFAULT 'Other',
    school_id INT,
    FOREIGN KEY (school_id) REFERENCES School(school_id) ON DELETE CASCADE
);

CREATE TABLE HospitalStaff(
    staff_id INT AUTO_INCREMENT PRIMARY KEY,
    hospital_coordinator_id INT,
    staff_profile LONGBLOB,
    firstname VARCHAR(50),
    middlename VARCHAR(50),
    lastname VARCHAR(50),
    email VARCHAR(50) UNIQUE NOT NULL,
    phone_number UNIQUE,
    birthdate DATE,
    gender ENUM('Male', 'Female', 'Other') DEFAULT 'Other',
    role TEXT,
    FOREIGN KEY (hospital_coordinator_id) REFERENCES HospitalCoordinator(hospital_coordinator_id) ON DELETE CASCADE
);


CREATE TABLE Appointment (
    appointment_id INT AUTO_INCREMENT PRIMARY KEY,
    hospital_coordinator_id INT NOT NULL,
    appointment_date DATETIME NOT NULL,
    status ENUM('Pending','Approved','Rejected','Completed','Cancelled') DEFAULT 'Pending',
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (hospital_coordinator_id) REFERENCES HospitalCoordinator(hospital_coordinator_id) ON DELETE CASCADE
);

CREATE TABLE AppointmentStudent (
    appointment_student_id INT AUTO_INCREMENT PRIMARY KEY,
    appointment_id INT NOT NULL,
    student_id INT NOT NULL,
    FOREIGN KEY (appointment_id) REFERENCES Appointment(appointment_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES Student(student_id) ON DELETE CASCADE,
    UNIQUE (appointment_id, student_id)
);

CREATE TABLE Billing (
    billing_id INT AUTO_INCREMENT PRIMARY KEY,
    appointment_id INT NOT NULL,
    school_id INT NOT NULL,
    total_students INT NOT NULL DEFAULT 0,
    total_amount DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    status ENUM('Unpaid','Paid','Partial') DEFAULT 'Unpaid',
    issued_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (appointment_id) REFERENCES Appointment(appointment_id) ON DELETE CASCADE,
    FOREIGN KEY (school_id) REFERENCES School(school_id) ON DELETE CASCADE
);

CREATE TABLE BillingItem (
    item_id INT AUTO_INCREMENT PRIMARY KEY,
    billing_id INT NOT NULL,
    description VARCHAR(255) NOT NULL,  -- e.g. Consultation, Lab Test
    amount DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (billing_id) REFERENCES Billing(billing_id) ON DELETE CASCADE
);
